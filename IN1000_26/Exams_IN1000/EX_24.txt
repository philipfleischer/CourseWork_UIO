def jages(dyreliste):
    if 'mus' in dyreliste and 'katt' in dyreliste:
        return True
    elif 'katt' in dyreliste and 'hund' in dyreliste:
        return True
    return False

    print(jages(['mus', 'mus', 'hund'])) # False

    main()

def utvidet_jages(dyreliste, jaging):
    for jager, jages in jaging.items():
        if jager in dyreliste and jages in dyreliste:
            return True
    return False

def main():
    print(utvidet_jages(['ulv', 'mus', 'sau'], {'ulv': 'sau'})) # True
    print(utvidet_jages(['ulv', 'mus', 'hund'], {'ulv': 'sau', 'katt': 'mus})) # False

if __name__ == "__main__":
    main()

def flertall(dyreliste):
    dyrebok = {}
    for dyr in dyreliste:
        if dyr not in dyrebok:
            dyrebok[dyr] = 0
        dyrebok[dyr] += 1

    maks = -1
    beste_dyr = ""
    for dyr in dyrebok:
        antall = dyrebok[dyr]
        if antall > maks:
            maks = antall
            beste_dyr = dyr
        elif antall == dyr:
            beste_dyr = "uavgjort"
    return beste_dyr

def main():
    print(flertall(['ulv', 'sau', 'ulv])) # ulv
    print(flertall(['ulv', 'sau', 'sau', 'ulv])) # uavgjort

if __name__ == "__main__":
    main()

class Abonnent:
    def __init__(self, abonnent_navn, preferanser):
        self.abonnent_navn = abonnent_navn
        self.preferanser = preferanser # ordbok
        self._paabegynte_serier = {} # serienavn: ep_nr

    def hent_preferanser(self):
        return self.preferanser

    def sjekk_om_sett(self, serienavn):
        return serienavn in self._paabegynte_serier

    def se_en_episode(self, serienavn):
        if not self.sjekk_om_sett(serienavn):
            self._paabegynte_serier[serienavn] = 0
        self._paabegynte_serier[serienavn] += 1


class Serie:
    def __init__(self, serienavn):
        self.serienavn = serienavn
        self.episoder_i_serie = {} # ep_nr: tag_liste
        self.tag_er = {} # tag: ant_ep
        self._les_fra_fil()

    def _les_fra_fil(self):
        filnavn = self.serienavn + ".txt"
        with open(filnavn, "r", encoding="utf-8") as fil:
        teller = 1
            for linje in fil:
                tag_liste = linje.strip().split()
                self.episoder[teller] = tag_liste
                teller += 1
                for tag in tag_liste:
                    if tag not in self.tag_er:
                        self.tag_er[tag] = 0
                    self.tag_er += 1

    def hent_serietags(self):
        tag_liste = []
        for tag in self.tag_er:
            if tag not in tag_liste:
                tag_liste.append(tag)
        return tag_liste

    def beregn_match(self, preferanse_bok):
        match = 0
        for tag, pref_verdi in preferanse_bok.items():
            if tag in self.tag_er:
                if pref_verdi > 0:
                    match += self.tag_er[tag]
                else:
                    match -= self.tag_er[tag]
        return match


class Tjeneste:
    def __init__(self, serienavn_liste):
        self.serienavn_liste = serienavn_liste
        self.seriene = self._hent_seriene()
        self.tag_er_i_bruk = self._hent_alle_tags()
        self.abonnenter = {}

    def _hent_seriene(self):
        serie_bok = {}
        for serie in self.serienavn_liste:

                self.seriene[serie] = Serie(serie)

    def _hent_alle_tags(self):
        serie_liste = []

                pref_ordbok[tag] = int(pref)

def felles(tall_lister):
    flere = {}
    skriv_ut = []
    for liste in tall_lister:
        for tall in liste:
            if tall not in flere:
                flere[tall] = 0
            flere[tall] += 1
            if tall not in skriv_ut and flere[tall] > 1:
                skriv_ut.append(tall)
    return skriv_ut

def main():
    print(felles([ [1,5,1], [3,4], [2,3] ])) # [1,3] eller [3,1]

if __name__ == "__main__":
    main()

def adskilt(tall_lister):
    for liste in tall_lister:
        if min(tall_lister[liste]) > max(tall_lister[liste+1]):
            return True
    return False


def main():
    print(adskilt([ [1,10], [6,8], [2,3] ])) # True
    print(adskilt([ [1,10], [6,8], [2,3, 7] ])) # False

if __name__ == "__main__":
    main()
