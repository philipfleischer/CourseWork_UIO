    def hent_ant_antrekk(self):
        return self._ant_trekk_med_i

    def oppdater_ant_antrekk(self, antall):

class Antrekk:
    def __init__(self, plagg_liste, anledning_liste):
        self.plagg_liste = plagg_liste
        self.anledning_liste = anledning_liste
        for plagg in self.plagg_liste:
            plagg.oppdater_ant_antrekk(1)

    def hent_plaggene(self):
        return self.plagg_liste

    def legg_til_anledning(self, anledning):
        if anledning not in self.anledning_liste:
            self.anledning_liste.append(anledning)

    def passer_til(self, anledning):
        return anledning in self.anledning_liste

    def har_farge(self, farge):
        for plagg in self.plagg_liste:
            if plagg.har_farge(farge):
                return True
        return False
