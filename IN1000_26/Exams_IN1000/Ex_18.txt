def vinnerlag(hjemmelag, bortelag, hjemmemaal, bortemaal):
    if hjemmemaal == bortemaal:
        return "uavgjort"
    if hjemmemaal > bortemaal:
        return hjemmelag
    return bortelag

def main():
    print(vinnerlag("Brann", "Molde", 2, 3))    # "Molde"
    print(vinnerlag("Brann", "Molde", 2, 2))    # "uavgjort"

if __name__ == "__main__":
    main()

def gull(lagoversikt):
    max_lag = None
    for lag, poeng in lagoversikt.items():
        if max_lag is None:
            max_lag = lag
        if lagoversikt[max_lag] < poeng:
            max_lag = lag
    return max_lag

def main():
    print(gull({"Brann" : 2, "Molde" : 3})) # "Molde

if __name__ == "__main__":
    main()

class Rett:
    def __init__(self, navn, pris, innholdsstoffer):
        self._navn = navn
        self._pris = pris
        self._innholdsstoffer = innholdsstoffer

    def sjekkInnholdOK(self, innholds_liste):
        for stoff in innholds_liste:
            if stoff in self._innholdsstoffer:
                return False
        return True

    def __str__(self):
        utskrift = f"{self._navn} koster: {self._pris} og inneholder:\n"
        for stoff in self._innholdsstoffer:
            utskrift += f"  {stoff}\n"
        return utskrift

class Kategori:
    def __init__(self, kategori_navn, retter):
        self._kategori_navn = kategori_navn
        self._retter = retter

    def hentOkRetter(self, innholds_liste):
        ok_liste = []
        for rett in self._retter:
            if rett.sjekkInnholdOK(innholds_liste):
                ok_liste.append(rett)
        return ok_liste

    def hent_retter(self):
        return self._retter

    def __len__(self):
        return len(self._retter)


class Meny:
    def __init__(self, kategori_navn):
        self._menyen = {}
        for kategori in kategori_navn:
            filnavn = kategori + ".txt"
            self._menyen[kategori] = self._lesKategoriFil(filnavn)

    def hentRedusertMeny(self, innholds_liste):
        redusert_meny = {}
        for navn, kategori in self._menyen.items():
            ok_kategori = Kategori(navn, kategori.hentOkRetter(innholds_liste))
            if len(ok_kategori) > 0:
                redusert_meny[navn] = ok_kategori
        return redusert_meny

def godkjenn(aldre):
    alle_familier = True
    for familie in aldre:
        ant_myndig = 0
        for a in familie:
            if a > 18:
                ant_myndig += 1
        if ant_myndig = 0:
            return False
    return True
